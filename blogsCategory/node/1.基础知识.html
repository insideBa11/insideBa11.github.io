# qs模块（querystring——字符串）

用途：拼接url里的参数

let qs = require('querystring')
appid=APPID&secret=SECRET&js_code=JSCODE&grant_type=authorization_code
const param = qs.stringify({
    'appid': 123,
    'secret': '123',
    'js_code': 123,
    'grant_type': 123
})

let url = 'https://api.weixin.qq.com/sns/jscode2session?' + param;
console.log(url); // https://api.weixin.qq.com/sns/jscode2session?appid=123&secret=123&js_code=123&grant_type=123

let test1 = qs.stringify({ foo: 'bar', baz: ['qux', 'quux'], corge: '' });
let test2 = qs.stringify({ foo: 'bar', baz: 'qux' }, ';', ':');
console.log(test1); // foo=bar&baz=qux&baz=quux&corge=
console.log(test2); // foo:bar;baz:qux
// 默认是 = 和 & ，可以在第二个参数修改 & ，第三个参数修改 =

# path模块（路径模块）

path.resolve() 是 Node.js 中 path 模块提供的一个方法，用于将路径解析为绝对路径。

const url = path.resolve(__dirname, './test.txt');
console.log(url); // /Users/xiejiawei/insideBall/宇宙爆炸的秘密/node/test.txt

# fs模块（文件系统模块）

const fs = require('fs');
const path = require('path')
const url = path.resolve(__dirname, './test.txt');

// 异步读取文件
fs.readFile(url, 'utf8', (err, data) => {
    if (err) throw err;
    console.log(data);
})
console.log(123);

// 同步读取文件
const str = fs.readFileSync(url, 'utf8')
console.log(str, 'daye');
console.log(456);
// 123
// muggle daye
// 456
// muggle
